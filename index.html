// 채점 결과 성적DB 저장 function saveGrades() { if (!currentExamSetup) { showAlert("명단과 세팅을 먼저 불러와 주세요."); return; } const campusEl = document.getElementById('inputCampus'); const campus = campusEl.options[campusEl.selectedIndex].text; const examName = document.getElementById('inputExamName').value.trim(); const courseName = document.getElementById('inputCourseName').value.trim(); const className = document.getElementById('inputClassName').value.trim(); // TODO: 추후 선생님 로그인/인증 연동 시 변경될 부분 const teacherName = "담임선생님"; const tbody = document.getElementById('inputTableBody'); const rows = tbody.querySelectorAll('tr'); let studentsData = []; rows.forEach((tr, rIdx) => { // 학생명 (2번째 td) let studentName = tr.cells[1].innerText.trim(); let totalScore = document.getElementById(`totalScore_${rIdx}`).innerText; // 해당 학생의 O/X 입력값 수집 (시작 번호부터 세팅된 문항 수만큼) let answers = {}; const inputs = tr.querySelectorAll('.grid-nav'); inputs.forEach((input, cIdx) => { let qNum = currentExamSetup.startNum + cIdx; let val = input.value; // 'O', 'X', 혹은 빈칸 answers[qNum] = val; }); studentsData.push({ name: studentName, school: "미상", // 현재 모의 데이터 기준 grade: "미상", // 현재 모의 데이터 기준 totalScore: totalScore, answers: answers }); }); if (studentsData.length === 0) { showAlert("저장할 학생 명단 데이터가 없습니다."); return; } const payload = { action: 'saveStudentGrades', campus: campus, examName: examName, courseName: courseName, className: className, teacherName: teacherName, students: studentsData }; showLoading(true, "채점 결과를 성적DB에 저장 중입니다..."); const url = document.getElementById('backendUrl').value; if (url.includes('YOUR_GAS_ID_HERE')) { setTimeout(() => { showLoading(false); showAlert("✔ [저장 성공 (로컬 모의)]
성적 데이터가 성공적으로 저장되었습니다."); }, 800); return; } fetch(url, { method: 'POST', body: JSON.stringify(payload) }) .then(res => res.json()) .then(result => { showLoading(false); if(result.success) showAlert("✔ 저장 완료
학생들의 채점 결과가 성적DB에 안전하게 저장되었습니다."); else showAlert("· Error: " + result.error); }) .catch(err => { showLoading(false); showAlert("통신 오류가 발생했습니다.
" + err.message); }); }
